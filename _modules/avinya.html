[docs]  def isButtonPressed(button):
    pot = ADC(Pin(BUTTON_ANALOG_PIN))
    pot.atten(ADC.ATTN_11DB)
    pot.width(ADC.WIDTH_12BIT)
    val = pot.read()
    if button == LEFT:
      return val > LEFT_BUTTON_THRESHOLD and val < RIGHT_BUTTON_THRESHOLD
    elif button == RIGHT:
      return val > RIGHT_BUTTON_THRESHOLD
    else:
      return False
